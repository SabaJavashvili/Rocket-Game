/* ------------------------------------------------------------------
   Starfield generator — draws a 2×-tall field of stars into each
   canvas so the CSS translateY(50%) loop is seamless.
   Three layers with different density, size, and brightness give
   us depth without any JS animation loop.
-------------------------------------------------------------------*/
(function () {
  const layers = [
    { el: document.getElementById('starsFar'),  count: 160, sizeMin: 0.4, sizeMax: 1.1, alphaMin: 0.25, alphaMax: 0.7,  tint: false },
    { el: document.getElementById('starsMid'),  count: 110, sizeMin: 0.8, sizeMax: 1.8, alphaMin: 0.45, alphaMax: 0.9,  tint: true  },
    { el: document.getElementById('starsNear'), count:  60, sizeMin: 1.4, sizeMax: 2.8, alphaMin: 0.75, alphaMax: 1.0,  tint: true  },
  ];

  // Subtle blue/violet/white tint pool (no saturated colors)
  const TINTS = [
    [233, 236, 255], // white
    [210, 220, 255], // cool white
    [200, 190, 255], // faint violet
    [255, 235, 220], // faint warm
  ];

  function drawLayer(layer) {
    const { el } = layer;
    // Canvas is 2× viewport height so the 50% translate wraps cleanly.
    const dpr = Math.min(window.devicePixelRatio || 1, 2);
    const w = window.innerWidth;
    const h = window.innerHeight * 2;
    el.width  = w * dpr;
    el.height = h * dpr;
    el.style.width  = w + 'px';
    el.style.height = h + 'px';
    const ctx = el.getContext('2d');
    ctx.scale(dpr, dpr);
    ctx.clearRect(0, 0, w, h);

    for (let i = 0; i < layer.count; i++) {
      const x = Math.random() * w;
      const y = Math.random() * h;
      const r = layer.sizeMin + Math.random() * (layer.sizeMax - layer.sizeMin);
      const a = layer.alphaMin + Math.random() * (layer.alphaMax - layer.alphaMin);
      const [cr, cg, cb] = layer.tint
        ? TINTS[Math.floor(Math.random() * TINTS.length)]
        : TINTS[0];

      // Soft star — tight radial glow
      const g = ctx.createRadialGradient(x, y, 0, x, y, r * 3);
      g.addColorStop(0,   `rgba(${cr},${cg},${cb},${a})`);
      g.addColorStop(0.4, `rgba(${cr},${cg},${cb},${a * 0.35})`);
      g.addColorStop(1,   `rgba(${cr},${cg},${cb},0)`);
      ctx.fillStyle = g;
      ctx.beginPath();
      ctx.arc(x, y, r * 3, 0, Math.PI * 2);
      ctx.fill();

      // Hard core
      ctx.fillStyle = `rgba(${cr},${cg},${cb},${Math.min(1, a + 0.2)})`;
      ctx.beginPath();
      ctx.arc(x, y, r * 0.6, 0, Math.PI * 2);
      ctx.fill();
    }
  }

  function draw() { layers.forEach(drawLayer); }

  // Debounced resize so we don't regenerate on every pixel
  let rid;
  window.addEventListener('resize', () => {
    clearTimeout(rid);
    rid = setTimeout(draw, 120);
  });
  draw();
})();

/* ------------------------------------------------------------------
   Ember particles — spawn periodically behind the rocket so the
   exhaust has follow-through / trailing hot bits.
-------------------------------------------------------------------*/
(function () {
  const host = document.getElementById('embers');
  if (!host) return;
  if (window.matchMedia('(prefers-reduced-motion: reduce)').matches) return;

  const MAX = 24;
  function spawn() {
    if (host.childElementCount >= MAX) return;
    const e = document.createElement('span');
    e.className = 'ember';
    const drift = (Math.random() - 0.5) * 60;        // horizontal sway
    const dur   = 1.2 + Math.random() * 1.1;         // life (s)
    const size  = 2 + Math.random() * 2.5;
    const warm  = Math.random() < 0.5;
    e.style.setProperty('--drift', drift + 'px');
    e.style.animationDuration = dur + 's';
    e.style.width  = size + 'px';
    e.style.height = size + 'px';

      e.style.background = 'var(--flame-cold)';
      e.style.boxShadow  = '0 0 6px var(--flame-cold), 0 0 14px #4a2bff';

  }
  setInterval(spawn, 90);

/* ------------------------------------------------------------------
   Keyboard flight controls
   Arrow keys or WASD move the rocket around the viewport with
   gentle acceleration + damping so it glides. Left/right also banks
   the rocket into the turn for a nicer feel.
   The pilot layer holds translate + rotate; .wobble / .vibe below
   it keep their idle jitter so the rocket still feels alive when idle.
-------------------------------------------------------------------*/
(function () {
  const pilot = document.getElementById('pilot');
  const hint  = document.getElementById('hint');
  if (!pilot) return;

  // State (in px, relative to the centered rocket rest position)
  let x = 0, y = 0;       // position offset
  let vx = 0, vy = 0;     // velocity
  let bank = 0;           // current bank angle (deg)

  // Expose live state + reset so other systems (lasers, UFOs, game) can read it.
  window.ROCKET = {
    get x()    { return x; },
    get y()    { return y; },
    get bank() { return bank; },
    get alive(){ return !window.GAME || !window.GAME.over; },
    // Direct input hook for touch controls — set any of
    // ROCKET.keys.up / down / left / right to true/false.
    keys: null, // set below after keys is defined
    // Tap-to-move: set {x, y} in rocket-offset space (0,0 = rest position)
    // and the flight loop will fly there. Set to null to release.
    target: null,
    reset() {
      x = 0; y = 0; vx = 0; vy = 0; bank = 0;
      for (const k in keys) keys[k] = false;
      window.ROCKET.target = null;
    },
  };

  // Tuning — touch devices fly faster toward tap targets, but with gentler
  // acceleration so the trip still feels deliberate.
  const IS_TOUCH = matchMedia('(pointer: coarse)').matches;
  const ACCEL    = IS_TOUCH ? 0.65 : 0.9;   // px / frame^2
  const MAX_V    = IS_TOUCH ? 18   : 14;    // max velocity (px / frame)
  const DAMP     = 0.90;  // per-frame velocity retention when no input
  const MAX_BANK = 18;    // deg
  const BANK_LERP= 0.12;  // how quickly bank catches up to target

  // Bounds — keep rocket within a safe margin of the viewport
  function bounds() {
    const margin = 60;
    return {
      minX: -window.innerWidth  / 2 + margin,
      maxX:  window.innerWidth  / 2 - margin,
      minY: -window.innerHeight / 2 + margin,
      maxY:  window.innerHeight / 2 - margin,
    };
  }

  // Input
  const keys = Object.create(null);
  window.ROCKET.keys = keys;
  const KEYMAP = {
    ArrowUp: 'up', ArrowDown: 'down', ArrowLeft: 'left', ArrowRight: 'right',
    w: 'up', W: 'up', s: 'down', S: 'down', a: 'left', A: 'left', d: 'right', D: 'right',
  };

  let hintTimer = null;
  function fadeHint() {
    if (!hint) return;
    clearTimeout(hintTimer);
    hintTimer = setTimeout(() => hint.classList.add('fade'), 2000);
  }
  fadeHint();

  window.addEventListener('keydown', (e) => {
    const dir = KEYMAP[e.key];
    if (!dir) return;
    keys[dir] = true;
    if (hint) hint.classList.add('fade');
    // Stop the page from scrolling on arrow keys.
    if (e.key.startsWith('Arrow') || e.key === ' ') e.preventDefault();
  }, { passive: false });

  window.addEventListener('keyup', (e) => {
    const dir = KEYMAP[e.key];
    if (dir) keys[dir] = false;
  });

  // Release keys when the tab loses focus so the rocket doesn't
  // keep accelerating into the void.
  window.addEventListener('blur', () => {
    for (const k in keys) keys[k] = false;
  });

  function tick() {
    // Apply thrust
    let ax = 0, ay = 0;

    // Target-seek mode: if window.ROCKET.target is set (x,y in rocket-offset
    // space, i.e. same coords as `x`/`y`), accelerate toward it. This is
    // used by the mobile touch-to-move controller — tap a point and the
    // rocket flies there fast. Falls back to key thrust when no target.
    const T = window.ROCKET && window.ROCKET.target;
    if (T) {
      const dx = T.x - x, dy = T.y - y;
      const d  = Math.hypot(dx, dy);
      if (d > 2) {
        const nx = dx / d, ny = dy / d;
        ax += nx * ACCEL * 1.6;  // a little punchier for tap-to-move
        ay += ny * ACCEL * 1.6;
      } else {
        // Arrived — clear the target so damping brings us to rest.
        window.ROCKET.target = null;
      }
    } else {
      if (keys.left)  ax -= ACCEL;
      if (keys.right) ax += ACCEL;
      if (keys.up)    ay -= ACCEL;
      if (keys.down)  ay += ACCEL;
    }

    vx += ax; vy += ay;

    // Damp when no input on that axis
    if (!T) {
      if (!keys.left && !keys.right) vx *= DAMP;
      if (!keys.up   && !keys.down)  vy *= DAMP;
    } else {
      // Light global damping while seeking, so we don't overshoot wildly.
      vx *= 0.96; vy *= 0.96;
    }

    // Clamp velocity
    const v2 = vx * vx + vy * vy;
    if (v2 > MAX_V * MAX_V) {
      const s = MAX_V / Math.sqrt(v2);
      vx *= s; vy *= s;
    }

    // Integrate position
    x += vx; y += vy;

    // Constrain to viewport
    const b = bounds();
    if (x < b.minX) { x = b.minX; vx = 0; }
    if (x > b.maxX) { x = b.maxX; vx = 0; }
    if (y < b.minY) { y = b.minY; vy = 0; }
    if (y > b.maxY) { y = b.maxY; vy = 0; }

    // Bank into the turn — target angle proportional to vx
    const targetBank = Math.max(-MAX_BANK, Math.min(MAX_BANK, (vx / MAX_V) * MAX_BANK));
    bank += (targetBank - bank) * BANK_LERP;

    pilot.style.transform =
      `translate(${x.toFixed(2)}px, ${y.toFixed(2)}px) rotate(${bank.toFixed(2)}deg)`;

    requestAnimationFrame(tick);
  }
  requestAnimationFrame(tick);
})();

  // ---- Main combat tick ----
  let lastFrameTs = 0;
  function tick(now) {
    // Frame-rate-independent step. Browsers (iOS Safari especially) may
    // throttle rAF to ~30Hz when idle and boost back to 60Hz during active
    // touch — that caused UFOs to visibly "accelerate" while the user was
    // dragging. Normalize every per-frame delta against a 60Hz baseline.
    let step = 1;
    if (lastFrameTs) {
      step = (now - lastFrameTs) / (1000 / 60);
      // Clamp to sane range — avoids a giant jump after a tab-switch.
      if (step < 0.2) step = 0.2;
      if (step > 3)   step = 3;
    }
    lastFrameTs = now;

    // Spawn UFOs
    if (!GAME.over && now >= nextSpawn) {
      spawnUFO();
      nextSpawn = now + UFO_MIN_INTERVAL + Math.random() * (UFO_MAX_INTERVAL - UFO_MIN_INTERVAL);
    }

    // Update bolts
    for (let i = bolts.length - 1; i >= 0; i--) {
      const b = bolts[i];
      b.x += b.vx * step; b.y += b.vy * step;
      b.el.style.left = b.x + 'px';
      b.el.style.top  = b.y + 'px';
      if (b.y < -40 || b.y > window.innerHeight + 40 ||
          b.x < -40 || b.x > window.innerWidth  + 40) {
        b.el.remove(); bolts.splice(i, 1);
      }
    }

    // Update UFOs
    for (let i = ufos.length - 1; i >= 0; i--) {
      const u = ufos[i];
      u.sway += 0.04 * step;
      u.x += (u.vx + Math.sin(u.sway) * u.swayAmp) * step;
      u.y += u.vy * step;
      u.el.style.left = u.x + 'px';
      u.el.style.top  = u.y + 'px';

      // Bolt collisions
      for (let j = bolts.length - 1; j >= 0; j--) {
        if (hitUFOvsBolt(u, bolts[j])) {
          const b = bolts[j];
          burst(b.x, b.y, { count: 8, hue: 'cool' });
          b.el.remove(); bolts.splice(j, 1);
          u.hp -= 1;
          if (u.hp <= 0) {
            const kx = u.x + u.w/2, ky = u.y + u.h/2;
            burst(kx, ky, { count: 22, hue: 'warm' });
            u.el.remove(); ufos.splice(i, 1);
            GAME.score += 1;
            if (scoreEl) {
              scoreEl.textContent = String(GAME.score);
              scoreEl.classList.add('bump');
              setTimeout(() => scoreEl.classList.remove('bump'), 160);
            }
            flyCoin(kx, ky);
            if (window.__onUFOKill) window.__onUFOKill(kx, ky);
            break;
          } else {
            u.el.classList.add('flash');
            setTimeout(() => u.el && u.el.classList.remove('flash'), 80);
          }
        }
      }
      if (!ufos[i]) continue;

      // Rocket collision → game over
      if (!GAME.over && hitRocketVsUFO(u)) {
        burst(u.x + u.w/2, u.y + u.h/2, { count: 30, hue: 'warm' });
        u.el.remove(); ufos.splice(i, 1);
        gameOver();
        continue;
      }

      // Offscreen cull (bottom)
      if (u.y > window.innerHeight + 40) {
        u.el.remove(); ufos.splice(i, 1);
      }
    }

    // Auto-fire while space held
    if (firing && !GAME.over) {
      if (now - lastShot >= COOLDOWN) { fire(); lastShot = now; }
    }

    requestAnimationFrame(tick);
  }

    "ჰაერში ავწიე ! ! ! 🌪️🚀",
    "გავაქრე?! 🌟🌟",
    "აბა ჰე ✌️💨",
    "ზუსტად მიზანში!!! 🎯💯",
    "ხოდა ეგრე!! 💪😤",
    "მტვრად ვაქციე... 🌫️",
    "ათიანში!!!!! 🌟🥇",
  ];

  // Avoid repeating the same quote twice in a row
  let lastQuoteIdx = -1;
  function pickQuote() {
    let i;
    do { i = Math.floor(Math.random() * KILL_QUOTES.length); }
    while (i === lastQuoteIdx && KILL_QUOTES.length > 1);
    lastQuoteIdx = i;
    return KILL_QUOTES[i];
  }

  function showKillQuote(x, y) {
    if (!quoteHost) return;
    const el = document.createElement('div');
    el.className = 'quote';
    el.textContent = pickQuote();
    // Anchor above the kill site, nudged into the viewport if near edges.
    const px = Math.max(120, Math.min(window.innerWidth - 120, x));
    const py = Math.max(60,  Math.min(window.innerHeight - 60, y - 44));
    el.style.left = px + 'px';
    el.style.top  = py + 'px';
    quoteHost.appendChild(el);

    // Animate: pop in, drift up, fade out.
    const DUR  = 1500;
    const start = performance.now();
    const rot  = (Math.random() - 0.5) * 8; // tiny tilt for personality
    function step(now) {
      const t = Math.min(1, (now - start) / DUR);
      // Pop + drift
      const scale = t < 0.15 ? (0.4 + (t / 0.15) * 0.7) : (1.1 - t * 0.1);
      const rise  = -40 * t;
      const op    = t < 0.1 ? t / 0.1 : (t > 0.75 ? (1 - t) / 0.25 : 1);
      el.style.transform =
        `translate(-50%, calc(-50% + ${rise}px)) rotate(${rot}deg) scale(${scale})`;
      el.style.opacity = String(op);
      if (t < 1) requestAnimationFrame(step);
      else el.remove();
    }
    requestAnimationFrame(step);
  }

  // Patch into the kill flow: we already bump score + burst in the UFO tick,
  // so we just hook an additional quote display + set up pickups loop via
  // a second rAF runner dedicated to falling coins.
  let lastPickupTs = 0;
  function pickupTick(now) {
    let step = 1;
    if (lastPickupTs) {
      step = (now - lastPickupTs) / (1000 / 60);
      if (step < 0.2) step = 0.2;
      if (step > 3)   step = 3;
    }
    lastPickupTs = now;

    if (!GAME.over && now >= nextPickup) {
      spawnPickup();
      nextPickup = now + PICKUP_MIN + Math.random() * (PICKUP_MAX - PICKUP_MIN);
    }
    for (let i = pickups.length - 1; i >= 0; i--) {
      const p = pickups[i];
      p.sway += 0.05 * step;
      p.x += (p.vx + Math.sin(p.sway) * 0.8) * step;
      p.y += p.vy * step;
      p.el.style.left = p.x + 'px';
      p.el.style.top  = p.y + 'px';

      if (!GAME.over && hitRocketVsPickup(p)) {
        // Award +5, fly a coin to the HUD, sparkle burst
        const cx = p.x + p.w / 2, cy = p.y + p.h / 2;
        burst(cx, cy, { count: 14, hue: 'warm' });
        GAME.score += 5;
        if (scoreEl) {
          scoreEl.textContent = String(GAME.score);
          scoreEl.classList.remove('bump');
          void scoreEl.offsetWidth;
          scoreEl.classList.add('bump');
          setTimeout(() => scoreEl.classList.remove('bump'), 200);
        }
        flyCoin(cx, cy);
        p.el.remove(); pickups.splice(i, 1);
        continue;
      }
      if (p.y > window.innerHeight + 40) {
        p.el.remove(); pickups.splice(i, 1);
      }
    }
    requestAnimationFrame(pickupTick);
  }
  requestAnimationFrame(pickupTick);

  // Also clear pickups on reset
  const _origReset = reset;
  // eslint-disable-next-line no-func-assign
  reset = function () {
    pickups.forEach(p => p.el.remove()); pickups.length = 0;
    if (quoteHost) quoteHost.innerHTML = '';
    nextPickup = performance.now() + 2500;
    _origReset();
  };

  // Expose a kill-quote hook and call it from the UFO destroy path by
  // monkey-patching flyCoin? Simpler: observe score changes on kill.
  // We instead wire directly — patch the game's tick via a MutationObserver
  // on the score element is overkill. Hook here:
  window.__onUFOKill = (kx, ky) => showKillQuote(kx, ky);
})();

/* ------------------------------------------------------------------
   Mobile touch controls — TAP-TO-MOVE.
   Touch anywhere on screen: the rocket flies to that point (fast).
   Drag your finger: the rocket follows it continuously.
   Fire button dispatches synthetic Space keydown/keyup so the existing
   combat code handles it unchanged.
-------------------------------------------------------------------*/
(function () {
  const stick    = document.getElementById('stick');     // full-screen capture surface
  const fireBtn  = document.getElementById('fireBtn');
  const hint     = document.getElementById('hint');
  if (!stick || !fireBtn) return;

  const R = window.ROCKET;
  if (!R) return;

  let pointerId = null;

  // Convert a viewport (client) coord into rocket-offset space.
  // The rocket's rest position is the viewport center, and ROCKET.x/ROCKET.y
  // are offsets from there.
  function setTargetFromClient(clientX, clientY) {
    const tx = clientX - window.innerWidth  / 2;
    const ty = clientY - window.innerHeight / 2;
    R.target = { x: tx, y: ty };
  }

  function onDown(e) {
    if (pointerId !== null) return;
    pointerId = e.pointerId;
    try { stick.setPointerCapture(pointerId); } catch (_) {}
    if (hint) hint.classList.add('fade');
    setTargetFromClient(e.clientX, e.clientY);
    e.preventDefault();
  }
  function onMove(e) {
    if (e.pointerId !== pointerId) return;
    setTargetFromClient(e.clientX, e.clientY);
    e.preventDefault();
  }
  function onUp(e) {
    if (e.pointerId !== pointerId) return;
    try { stick.releasePointerCapture(pointerId); } catch (_) {}
    pointerId = null;
    // Drop the target — rocket damps to a stop at its current position.
    R.target = null;
  }

  // Ultra-safety: if all document touches go away, release.
  function releaseIfAllTouchesGone() {
    if (pointerId === null) return;
    pointerId = null;
    R.target = null;
  }
  document.addEventListener('touchend', (e) => {
    if (e.touches.length === 0) releaseIfAllTouchesGone();
  }, true);
  document.addEventListener('touchcancel', (e) => {
    if (e.touches.length === 0) releaseIfAllTouchesGone();
  }, true);

  stick.addEventListener('pointerdown',   onDown);
  stick.addEventListener('pointermove',   onMove);
  stick.addEventListener('pointerup',     onUp);
  stick.addEventListener('pointercancel', onUp);
  stick.addEventListener('lostpointercapture', (e) => {
    if (e.pointerId === pointerId) { pointerId = null; R.target = null; }
  });
  // Global watchdogs — catch releases the stick missed due to gesture arbitration.
  window.addEventListener('pointerup',     onUp, true);
  window.addEventListener('pointercancel', onUp, true);
  window.addEventListener('pointermove',   (e) => {
    if (pointerId !== null && e.pointerId === pointerId) {
      setTargetFromClient(e.clientX, e.clientY);
    }
  }, true);

  // --- Fire button: dispatch synthetic Space events ---
  function fireDown(ev) {
    ev.preventDefault();
    ev.stopPropagation();
    fireBtn.classList.add('pressed');
    window.dispatchEvent(new KeyboardEvent('keydown', { key: ' ', code: 'Space', bubbles: true }));
  }
  function fireUp(ev) {
    if (ev) { ev.preventDefault(); ev.stopPropagation && ev.stopPropagation(); }
    fireBtn.classList.remove('pressed');
    window.dispatchEvent(new KeyboardEvent('keyup',   { key: ' ', code: 'Space', bubbles: true }));
  }
  fireBtn.addEventListener('pointerdown',   fireDown);
  fireBtn.addEventListener('pointerup',     fireUp);
  fireBtn.addEventListener('pointercancel', fireUp);
  fireBtn.addEventListener('pointerleave',  () => fireBtn.classList.contains('pressed') && fireUp());

  // Safety: release everything on visibility change / blur
  document.addEventListener('visibilitychange', () => {
    if (document.hidden) { R.target = null; fireUp(); pointerId = null; }
  });
  window.addEventListener('blur', () => { R.target = null; fireUp(); pointerId = null; });

  // Block multi-touch gestures (pinch/zoom, double-tap zoom) on the game surface
  document.addEventListener('gesturestart',  (e) => e.preventDefault());
  document.addEventListener('touchstart',    (e) => {
    if (!e.target.closest('.stick, .fire-btn')) e.preventDefault();
  }, { passive: false });
  document.addEventListener('touchmove',     (e) => e.preventDefault(), { passive: false });
})();
</script>
